# Sample Python Code: Data Analysis
import pandas as pd

# Create a DataFrame
data = {'Product': ['A', 'B', 'C'], 'Sales': [100, 150, 200]}
df = pd.DataFrame(data)

# Calculate total sales
total_sales = df['Sales'].sum()

# Output result
print(f"Total Sales: {total_sales}")
